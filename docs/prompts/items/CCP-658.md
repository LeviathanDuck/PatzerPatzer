# CCP-658 — Import Animation: Fast Replay of Master Games During Opponent Fetch

## Context

When a user clicks "Import Games" for opponent research, the view immediately switches to
the board (CCP-656). During the fetch, the board currently sits static at the start position
with only a progress bar showing. This prompt adds a fast game-replay animation: the board
plays through moves from the CCP-657 master games dataset while the fetch runs, giving the
user something engaging to watch.

**Dependency:** CCP-657 must be complete first (`src/showcase/masterGames.ts` must exist).

## Task

All changes go in `src/openings/view.ts` only.


### 3. Animation speed constant

```typescript
const ANIM_MOVE_MS = 600; // ms per move — tweak later
```

### 4. Start/stop functions

```typescript
function startImportAnimation(redraw: () => void): void {
  stopImportAnimation();
  _animGame    = randomMasterGame();
  _animMoveIdx = 0;
  // Parse starting position
  const setup = parseFen('rnbqkbnr/pppppppp/8/8/8/8/PPPPPPPP/RNBQKBNR w KQkq - 0 1');
  _animPos = setup.isOk ? Chess.fromSetup(setup.value).unwrap() : null;
  if (!_animPos || !_openingsCg) return;
  // Reset board to start
  _openingsCg.set({
    fen: 'rnbqkbnr/pppppppp/8/8/8/8/PPPPPPPP/RNBQKBNR w KQkq - 0 1',
    lastMove: [] as Key[],
    movable: { color: 'both', free: false, dests: new Map() },
    animation: { enabled: true },
  });
  scheduleNextAnimMove(redraw);
}

function stopImportAnimation(): void {
  if (_animTimer !== null) {
    clearTimeout(_animTimer);
    _animTimer = null;
  }
  _animGame   = null;
  _animMoveIdx = 0;
  _animPos    = null;
}

function scheduleNextAnimMove(redraw: () => void): void {
  _animTimer = setTimeout(() => {
    _animTimer = null;
    if (!isFetching() || !_animGame || !_animPos || !_openingsCg) return;

    const uciStr = _animGame.moves[_animMoveIdx];
    if (!uciStr) {
      // Game ended — pick a new random game and restart
      startImportAnimation(redraw);
      return;
    }

    const move = parseUci(uciStr);
    if (!move || !_animPos.isLegal(move)) {
      // Bad move — skip to next game
      startImportAnimation(redraw);
      return;
    }

    // Apply move and update board
    _animPos.play(move);
    const newFen = makeFen(_animPos.toSetup());
    const from = uciStr.slice(0, 2) as Key;
    const to   = uciStr.slice(2, 4) as Key;
    _openingsCg.set({
      fen:      newFen,
      lastMove: [from, to],
    });

    _animMoveIdx++;
    scheduleNextAnimMove(redraw);
  }, ANIM_MOVE_MS);
}
```

### 5. Wire animation to fetch lifecycle

In `renderFetchBar(redraw)` (added in CCP-656), add a side-effect hook to start the
animation when the bar first renders and the board is available. The cleanest place is
a Snabbdom `hook: { insert }` on the bar's root element — or, simpler: call
`startImportAnimation(redraw)` once when `_animTimer === null && _animGame === null &&
_openingsCg !== undefined` at the top of `renderFetchBar`.

```typescript
function renderFetchBar(redraw: () => void): VNode {
  // Kick off animation the first time this renders (or if it stopped unexpectedly).
  if (_animTimer === null && _animGame === null && _openingsCg) {
    startImportAnimation(redraw);
  }
  // ... rest of existing renderFetchBar vnode ...
}
```

When `isFetching()` becomes false (fetch complete), `stopImportAnimation()` should be
called before the tree build takes over. The right place is inside `syncOpeningsBoard`:
at the top of that function, if `!isFetching()` and `_animGame !== null`, call
`stopImportAnimation()`. This ensures the board is cleanly handed off when
`openCollection` triggers `syncOpeningsBoard` after the fetch.

## Notes on chessops API

- `Chess.fromSetup(setup).unwrap()` — may throw; wrap in try/catch
- `pos.isLegal(move)` — check before `pos.play(move)`
- `makeFen(pos.toSetup())` — produces standard FEN string
- `parseUci(str)` — returns `Move | undefined`
- Castling UCIs: `'e1g1'` (white kingside), `'e1c1'` (white queenside), etc.
